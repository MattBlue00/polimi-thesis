import pandas as pd
import numpy as np

# Status standardization
df['status'] = df['status'].replace({'s': 'sold', 'f': 'for_sale', '-': np.nan})

# Price cleaning
df['price'] = df['price'].replace('[\$,]', '', regex=True)
df['price'] = pd.to_numeric(df['price'].replace('-', np.nan))

# Bed/Bath cleaning
df['bed'] = pd.to_numeric(df['bed'].replace([9999, '-'], np.nan))
df['bath'] = pd.to_numeric(df['bath'].replace(['-'], np.nan))

# Acre_lot cleaning
df['acre_lot'] = pd.to_numeric(df['acre_lot'].replace('-', np.nan))
df.loc[df['acre_lot'] < 0, 'acre_lot'] = np.nan

# Street cleaning
df['street'] = df['street'].replace(['-', 'NaN'], np.nan)

# Date standardization
df['prev_sold_date'] = pd.to_datetime(df['prev_sold_date'], format='mixed')

# House_size cleaning
df['house_size'] = pd.to_numeric(df['house_size'].replace('-', np.nan))
